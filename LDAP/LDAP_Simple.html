cat <<'EOF' > /usr/local/sbin/xurupita-adduser.sh #!/bin/bash 6001 ola=$(sed -n '2p' $0) &> /dev/null tr=$(expr $ola + 1) echo "o nome do usuario" read user cat << EOFa > $user-grupo.ldif dn: cn=$user,ou=Grupos,dc=xurupita,dc=nl objectClass: posixGroup objectClass: top cn: $user userPassword: {crypt}x gidNumber: $ola EOFa cat << EOFb > $user.ldif dn: uid=$user,ou=Usuarios,dc=xurupita,dc=nl uid: $user cn: $user objectclass: account objectClass: posixAccount objectClass: top objectClass: shadowAccount userPassword: 123456 shadowLastChange: 15140 shadowMin: 0 shadowMax: 99999 shadowWarning: 7 loginShell: /bin/bash uidNumber: $ola gidNumber: $ola homeDirectory: /home/$user gecos: $user description: $user@xurupita.nl EOFb ldapadd -x -h localhost -D"cn=admin,dc=xurupita,dc=nl" -w '123456' -f $user-grupo.ldif sleep 1 ldapadd -x -h localhost -D"cn=admin,dc=xurupita,dc=nl" -w '123456' -f $user.ldif rm -rf $user.ldif $user-grupo.ldif sed 2s/$ola/$tr/ $0 -i EOF

cat << 'EOF' > /usr/local/sbin/xurupita-addgroup.sh

#!/bin/bash

ola=$(sed -n '2p' /usr/local/sbin/xurupita-adduser.sh) &> /dev/null

tr=$(expr $ola + 1)

echo 'nome do grupo'

read grup

cat << EOFc > $grup.ldif

dn: cn=$grup,ou=Grupos,dc=xurupita,dc=nl

objectClass: posixGroup

objectClass: top

cn: $grup

userPassword: {crypt}x

gidNumber: $ola

EOFc

ldapadd -x -D cn=admin,dc=xurupita,dc=nl -w '123456' -f $grup.ldif

rm -rf $grup.ldif

sed 2s/$ola/$tr/ /usr/local/sbin/xurupita-adduser.sh -i

EOF

1

chmod +x /usr/local/sbin/xurupita-addgroup.sh
